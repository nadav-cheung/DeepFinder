- [ ] **Step 4: Add type-to-select logic to ResultsListState**

In `Sources/GUI/ResultsListView.swift`, add to `ResultsListState` (after `moveSelectionToGroup`, around line 204):

```swift
/// REQ-3.2-08: Type-to-select jumps to first file starting with the given character.
/// Pressing the same character repeatedly cycles through matching results.
func typeSelect(character: Character) {
    let count = min(visibleCount, allResults.count)
    guard count > 0 else { return }

    let target = Character(character.lowercased())
    let results = visibleResults

    // If already on a matching result, find the next one (cycle)
    let startIndex: Int
    if let current = selectedIndex {
        // Check if next result also matches (for cycling)
        let nextIndex = current + 1
        if nextIndex < count,
           results[nextIndex].record.originalName.first?.lowercased() == String(target) {
            startIndex = nextIndex
        } else {
            startIndex = 0
        }
    } else {
        startIndex = 0
    }

    for i in startIndex..<count {
        if results[i].record.originalName.first?.lowercased() == String(target) {
            selectedIndex = i
            return
        }
    }

    // Wrap: search from beginning
    if let current = selectedIndex, current > 0 {
        for i in 0..<min(current, count) {
            if results[i].record.originalName.first?.lowercased() == String(target) {
                selectedIndex = i
                return
            }
        }
    }
}
```

```swift
// Sources/GUI/BottomActionBar.swift
import SwiftUI

// MARK: - BottomActionBar

/// Fixed footer showing keyboard shortcuts for the selected result.
///
/// REQ-3.2-37: Replaces KeyboardHintBar with a context-aware action bar
/// that shows relevant shortcuts when a result is selected, and a default
/// hint when nothing is selected.
///
/// Follows the Raycast pattern: shortcut symbols in monospaced font,
/// action labels in secondary color, ~24pt height.
struct BottomActionBar: View {

    let hasSelection: Bool

    // MARK: - Hint Model

    private struct Hint {
        let symbol: String
        let label: String
    }

    // MARK: - Hints

    private let selectedHints: [Hint] = [
        Hint(symbol: "↵", label: "打开"),
        Hint(symbol: "⌘↵", label: "Finder"),
        Hint(symbol: "Space", label: "预览"),
        Hint(symbol: "⌘C", label: "路径"),
    ]

    private let defaultHint: String = "↑↓ 导航  ↵ 打开  Esc 关闭"

    // MARK: - Body

    var body: some View {
        HStack(spacing: 0) {
            if hasSelection {
                ForEach(Array(selectedHints.enumerated()), id: \.offset) { index, hint in
                    if index > 0 {
                        Rectangle()
                            .fill(.separator)
                            .frame(width: 1, height: 10)
                            .padding(.horizontal, 6)
                    }

                    HStack(spacing: 3) {
                        Text(hint.symbol)
                            .monospaced()
                        Text(hint.label)
                    }
                }
            } else {
                Text(defaultHint)
            }
        }
        .font(.system(size: 11))
        .foregroundStyle(.secondary)
        .padding(.vertical, 4)
        .padding(.horizontal, 12)
        .frame(height: 24)
    }
}
```

```swift
// Sources/GUI/PathShortener.swift
import Foundation

enum PathShortener {

    /// Maximum display length for a path.
    private static let maxLength = 40

    /// Shortens a path for display: replace home with ~, truncate middle with …
    static func shorten(_ path: String) -> String {
        let home = FileManager.default.homeDirectoryForCurrentUser.path
        var display = path
        if display.hasPrefix(home) {
            display = "~" + display.dropFirst(home.count)
        }

        guard display.count > maxLength else { return display }

        let prefixEnd = display.index(display.startIndex, offsetBy: 8)
        let suffixStart = display.index(display.endIndex, offsetBy: -min(20, display.count - 11))
        return String(display[..<prefixEnd]) + "…/" + String(display[suffixStart...])
    }
}
```
